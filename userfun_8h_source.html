139/*
140void get_sensors(void) gets the sensor readings
141*/
142void get_sensors(void) {
143 //imu
144 if (counters.imu == 0) {
145 imu.ACC_GetAxes(imuData_raw.accs_mg);
146 imu.GYRO_GetAxes(imuData_raw.gyros_mdps);
147 counters.imu = IMU_SAMPLING_FAC - 1;
148 }
149 else counters.imu--;
150
151 //mag
152 if (counters.mag == 0) {
153 mag.GetAxes(imuData_raw.mags_mG);
154 counters.mag = MAG_SAMPLING_FAC - 1;
155 }
156 else counters.mag--;
157
158 //speed
159 if (counters.speedSensor == 0) {
160 if (speedSensor.available()) speedSensor.rxObj(speed_raw);
161 counters.speedSensor = SPEED_SAMPLING_FAC - 1;
162 }
163 else counters.speedSensor--;
164
165 //gps
166 gps.read();
167 if (gps.newNMEAreceived()) {
168 gps.parse(gps.lastNMEA());
169 gpsData_raw.fix = gps.fix;
170 if (gpsData_raw.fix) {
171 gpsData_raw.lat = gps.latitudeDegrees;
172 gpsData_raw.lon = gps.longitudeDegrees;
173 gpsData_raw.speed = gps.speed*GPSSPEED_SCALE;
174 if (gps.lat == 'S') gpsData_raw.lat *= -1;
175 if (gps.lon == 'W') gpsData_raw.lon *= -1;
176 }
177 else {
178 gps.lat = NAN;
179 gps.lon = NAN;
180 gps.speed = NAN;
181 }
182 }
183
184 //analogs
185 actCurr_raw.actCurr = analogRead(ACT_CURR);
186 riderTorque_raw.riderTorque = analogRead(RIDER_TRQ);
187 steer_raw.steerPos = analogRead(STEER_POS);
188 steer_raw.steerVel = analogRead(STEER_SPEED);
189 voltage_raw.batVolt = analogRead(VOLTAGE);
190 forkDisp_raw.forkDisp = analogRead(FORK_DISP);
191
192 //sbus
193 if (sbus.Read()) {
194 if (check_sbus()) { //success!!!
195 sbus.ch(remote_raw.ch, NUM_CH_SBUS*sizeof(int16_t)); //read channels
196 remote_raw.missing_frame = 0; //reset missing frame
197 } else remote_raw.missing_frame++; //increment missing frame
198 } else remote_raw.missing_frame++; //increment missing frame
199 if (remote_raw.missing_frame >= MAX_MISSING_SBUS) { //set zero values to
200 remote_raw.missing_frame = MAX_MISSING_SBUS; //set to #MAX_MISSING_SBUS to avoid overflow
201 for (uint8_t i = 0; i < NUM_CH_SBUS; ++i) remote_raw.ch[i] = ZERO_SBUS; //set channel to zero
202 }
203}

205/*
206bool check_sbus(void) check for SBUS errors
207*/
208bool check_sbus(void) {
209 if (sbus.lost_frame()) { return false; }
210 if (sbus.failsafe()) { return false; }
211 for (uint8_t i = 0; i < NUM_CH_SBUS; ++i) {
212 if ((sbus.ch(i) < MIN_SBUS) || (sbus.ch(i) > MAX_SBUS)) { return false; } //return false as soon as an unfeasible value is found
213 }
214 return true; //all checks ok
215}

217/*
218void do_control(void) sets the control input structure for next step and perform control update cycle
219*/
220void do_control(void) {
221 //set speed
222 ctrl.controlModel_U.speed = SPEED_SCALE*float(speed_raw.speed);
223 ctrl.controlModel_U.dist = DIST_SCALE * float(speed_raw.dist);
224 //set the threee components of accs, gyros, mags
225 for (int i = 0; i < 3; i++) {
226 ctrl.controlModel_U.accs[i] = float(imuData_raw.accs_mg[0] * imu_rotMat[i][0] +
227 imuData_raw.accs_mg[1] * imu_rotMat[i][1] +
228 imuData_raw.accs_mg[2] * imu_rotMat[i][2]) * ACC_SCALE;
229 ctrl.controlModel_U.gyros[i] = ((float(imuData_raw.gyros_mdps[0]) - imuData_raw.gyros_offset[0]) * imu_rotMat[i][0] +
230 (float(imuData_raw.gyros_mdps[1]) - imuData_raw.gyros_offset[1]) * imu_rotMat[i][1] +
231 (float(imuData_raw.gyros_mdps[2]) - imuData_raw.gyros_offset[2]) * imu_rotMat[i][2] ) * GYRO_SCALE; //may have also offset
232 ctrl.controlModel_U.mags[i] = float(imuData_raw.mags_mG[0] * mag_rotMat[i][0] +
233 imuData_raw.mags_mG[1] * mag_rotMat[i][1] +
234 imuData_raw.mags_mG[2] * mag_rotMat[i][2]) * MAG_SCALE;
235 }
236 //set gps
237 ctrl.controlModel_U.gps[0] = gpsData_raw.lat;
238 ctrl.controlModel_U.gps[1] = gpsData_raw.lon;
239 ctrl.controlModel_U.gps[2] = gpsData_raw.speed;
240 //set steer position and speed
241 ctrl.controlModel_U.steer[0] = 0;
242 ctrl.controlModel_U.steer[1] = 0;
243 //set front fork displacement
244 ctrl.controlModel_U.forkdisp = CONVERT_FORKDISP_TO_MM(float(forkDisp_raw.forkDisp)) - forkDisp_raw.forkDisp_offset;
245 //set rider torque
246 ctrl.controlModel_U.torque = CONVERT_RIDERTRQ_TO_NM(float(riderTorque_raw.riderTorque)) - riderTorque_raw.riderTorque_offset;
247 //set steer position and velocity
248 ctrl.controlModel_U.steer[0] = CONVERT_STEERPOS_TO_RAD(float(steer_raw.steerPos));
249 ctrl.controlModel_U.steer[1] = CONVERT_STEERVEL_TO_RADPS(float(steer_raw.steerVel)) - steer_raw.steerVel_offset;
250 //set actual motor current
251 ctrl.controlModel_U.curr = CONVERT_ACTCURR_TO_A(float(actCurr_raw.actCurr)) - actCurr_raw.actCurr_offset;
252 //set battery voltage
253 ctrl.controlModel_U.voltage = CONVERT_VOLTAGE_TO_V(float(voltage_raw.batVolt)) - voltage_raw.batVolt_offset;
254 //CPU temp
255 ctrl.controlModel_U.CPUTemp = tempmonGetTemp();
256 //set remote control (2 channels) sbus TODO
257 ctrl.controlModel_U.ref_inputs[0] = CONVERT_CHANNEL_TO_FLOAT(remote_raw.ch[SBUS_ROLL_CH-1], MIN_REF_INPUT, MAX_REF_INPUT);
258 ctrl.controlModel_U.ref_inputs[1] = CONVERT_CHANNEL_TO_FLOAT(remote_raw.ch[SBUS_THROTTLE_CH-1], MIN_REF_INPUT, MAX_REF_INPUT);
259
260 //update control
261 ctrl.update(); //this performs the control loop
262}

281//set drivers
282void set_driver(void) {
283 bool enable = remote_raw.ch[SBUS_EN_CH-1] >= TRESHOLD_LOGIC_SBUS; //enable from remote control.
284
285 //set motor driver
286#if MTR_CTRL_MODE==0
287 analogWrite(PWM_PIN, constrain(CONVERT_CURRENT_TO_PWM(ctrl.controlModel_Y.curr_ref), PWM_MIN*(powf(2, PWM_RES) - 1), PWM_MAX*(powf(2, PWM_RES) - 1)));
288#else
289 dac.analogWrite(DAC_MTR_CH, constrain(CONVERT_CURRENT_TO_DAC(ctrl.controlModel_Y.curr_ref), 0, pow(2, DAC_RES) - 1));
290#endif
291 digitalWriteFast(MTR_DIR_PIN, (ctrl.controlModel_Y.curr_ref >= 0) ? CW : !CW);
292 digitalWriteFast(MTR_EN_PIN, (enable) ? MTR_EN_STATE : !MTR_EN_STATE);
293
294 //set throttle signal
295 dac.analogWrite(DAC_THROTTLE_CH, (ctrl.controlModel_Y.curr_ref >= 0) ?
296 enable*constrain(CONVERT_TRHOTTLE_TO_DAC(ctrl.controlModel_Y.throttle_ref), 0, pow(2, DAC_RES) - 1) :
297 CONVERT_TRHOTTLE_TO_DAC(0));
298 brakeMotor.moveAbsAsync((ctrl.controlModel_Y.curr_ref <= 0) ?
299 enable*constrain(CONVERT_TRHOTTLE_TO_STEPS(-ctrl.controlModel_Y.throttle_ref),0,MAX_BR_DISP) :
300 CONVERT_TRHOTTLE_TO_STEPS(0));
301
302 //set other stuff below
303}
